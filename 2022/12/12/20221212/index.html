Spring Boot中操作MongoDB

发表于2022-12-12|更新于2024-01-23|后端
|阅读量:

MongoDB

Mongodb是为快速开发互联网Web应用而构建的数据库系统，其数据模型和持久化策略就是为了构建高读/写吞吐量和高自动灾备伸缩性的系统。

Spring Data Mongodb

Spring Data Mongodb是Spring提供的一种以Spring Data风格来操作数据存储的方式，它可以避免编写大量的样板代码。

pom文件引用

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-data-mongodb</artifactId>
</dependency>

配置信息

application.properties配置Mongodb的连接信息

# spring.data.mongodb.uri=mongodb://user:password@ip:port/database
spring.data.mongodb.uri=mongodb://springbucks:springbucks@localhost:27017/springbucks

注解

  • @Document:标示映射到Mongodb文档上的领域对象
  • @Id:标示某个域为ID域
  • @Indexed:标示某个字段为Mongodb的索引字段
  • @Field:字段标识，指定值为字段名称
// 实体类
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;
import org.joda.money.Money;
import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.mapping.Document;

@Document
@Data
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Coffee {
    @Id
    private String id;
    private String name;
    private Money price;
    private Date createTime;
    private Date updateTime;
}

MongoTemplate

@Autowired
private MongoTemplate mongoTemplate;

@Override
public void run(ApplicationArguments args) throws Exception {
    // 单个保存 save
    Coffee espresso = Coffee.builder()
            .name("espresso")
            .price(Money.of(CurrencyUnit.of("CNY"), 20.0))
            .createTime(new Date())
            .updateTime(new Date()).build();
    Coffee saved = mongoTemplate.save(espresso);
    log.info("Coffee {}", saved);

    // 单个或批量保存 insert
    List<Coffee> coffees = new ArrayList<>(3);
    for(int i = 0; i <= 3; i++) {
        Coffee coffee = Coffee.builder()
                .name("espresso_" + i)
                .price(Money.of(CurrencyUnit.of("CNY"), 20.0))
                .createTime(new Date())
                .updateTime(new Date()).build();
        coffees.add(coffee);
    }
    List<Coffee> savedList = (List<Coffee>) mongoTemplate.insert(coffees, Coffee.class);
    log.info("CoffeeList {}", savedList);

    // find、Query 和 Criteria 实现查询
    List<Coffee> list = mongoTemplate.find(
            Query.query(Criteria.where("name").is("espresso")), Coffee.class);
    log.info("Find {} Coffee", list.size());
    list.forEach(c -> log.info("Coffee {}", c));

    // 为了看更新时间
    Thread.sleep(1000); 

    // update、Query 和 Criteria 实现更新
    UpdateResult result = mongoTemplate.updateFirst(query(where("name").is("espresso")),
            new Update().set("price", Money.ofMajor(CurrencyUnit.of("CNY"), 30))
                    .currentDate("updateTime"),
            Coffee.class);
    log.info("Update Result: {}", result.getModifiedCount());
    Coffee updateOne = mongoTemplate.findById(saved.getId(), Coffee.class);
    log.info("Update Result: {}", updateOne);

    // 删除 remove
    mongoTemplate.remove(updateOne);
    }

MongoRepository

添加@EnableMongoRepositories注解
通过继承MongoRepository接口，可以实现对一个对象的增删改查。
首先继承MongoRepository<T, ID> 接口，其中T为bean类，ID为该bean的唯一标识的类型，一般为ObjectId或者String。

// Repository类
import geektime.spring.data.mongodemo.model.Coffee;
import org.springframework.data.mongodb.repository.MongoRepository;

import java.util.List;

public interface CoffeeRepository extends MongoRepository<Coffee, String> {
    // 增加额外的查询方法，findBy+字段名+方法后缀即可，参数为字段值
    List<Coffee> findByName(String name);
}

@Autowired
private CoffeeRepository coffeeRepository;

@Override
public void run(String... args) throws Exception {
    Coffee espresso = Coffee.builder()
            .name("espresso")
            .price(Money.of(CurrencyUnit.of("CNY"), 20.0))
            .createTime(new Date())
            .updateTime(new Date()).build();
    Coffee latte = Coffee.builder()
            .name("latte")
            .price(Money.of(CurrencyUnit.of("CNY"), 30.0))
            .createTime(new Date())
            .updateTime(new Date()).build();

    // insert支持单个和批量保存
    coffeeRepository.insert(Arrays.asList(espresso, latte));
    // findAll查询所有符合条件的数据；Sort排序
    coffeeRepository.findAll(Sort.by("name"))
            .forEach(c -> log.info("Saved Coffee {}", c));

    Thread.sleep(1000);
    latte.setPrice(Money.of(CurrencyUnit.of("CNY"), 35.0));
    latte.setUpdateTime(new Date());
    // save更新数据
    coffeeRepository.save(latte);
    // 根据字段查询
    coffeeRepository.findByName("latte")
            .forEach(c -> log.info("Coffee {}", c));

    // 删除所有
    coffeeRepository.deleteAll();
}
